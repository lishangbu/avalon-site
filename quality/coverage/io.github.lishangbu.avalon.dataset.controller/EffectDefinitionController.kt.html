package io.github.lishangbu.avalon.dataset.controller

import io.github.lishangbu.avalon.dataset.entity.dto.EffectDefinitionSpecification
import io.github.lishangbu.avalon.dataset.entity.dto.EffectDefinitionView
import io.github.lishangbu.avalon.dataset.entity.dto.SaveEffectDefinitionInput
import io.github.lishangbu.avalon.dataset.entity.dto.UpdateEffectDefinitionInput
import io.github.lishangbu.avalon.dataset.service.EffectDefinitionService
import jakarta.validation.Valid
import org.babyfish.jimmer.Page
import org.springframework.data.domain.Pageable
import org.springframework.web.bind.annotation.DeleteMapping
import org.springframework.web.bind.annotation.GetMapping
import org.springframework.web.bind.annotation.ModelAttribute
import org.springframework.web.bind.annotation.PathVariable
import org.springframework.web.bind.annotation.PostMapping
import org.springframework.web.bind.annotation.PutMapping
import org.springframework.web.bind.annotation.RequestBody
import org.springframework.web.bind.annotation.RequestMapping
import org.springframework.web.bind.annotation.RestController

/** 战斗效果定义控制器。 */
@RestController
@RequestMapping("/effect-definition")
class EffectDefinitionController(
    /** 战斗效果定义应用服务。 */
    private val effectDefinitionService: EffectDefinitionService,
) {
    /** 分页查询效果定义。 */
    @GetMapping("/page")
    fun getEffectDefinitionPage(
        pageable: Pageable,
        @ModelAttribute specification: EffectDefinitionSpecification,
    ): Page<EffectDefinitionView> = effectDefinitionService.getPageByCondition(specification, pageable)

    /** 查询效果定义列表。 */
    @GetMapping("/list")
    fun listEffectDefinitions(
        @ModelAttribute specification: EffectDefinitionSpecification,
    ): List<EffectDefinitionView> = effectDefinitionService.listByCondition(specification)

    /** 创建效果定义。 */
    @PostMapping
    fun save(
        @Valid
        @RequestBody command: SaveEffectDefinitionInput,
    ): EffectDefinitionView = effectDefinitionService.save(command)

    /** 更新效果定义。 */
    @PutMapping
    fun update(
        @Valid
        @RequestBody command: UpdateEffectDefinitionInput,
    ): EffectDefinitionView = effectDefinitionService.update(command)

    /** 删除指定 ID 的效果定义。 */
    @DeleteMapping("/{id:\\d+}")
    fun deleteById(
        @PathVariable id: Long,
    ) {
        effectDefinitionService.removeById(id)
    }
}

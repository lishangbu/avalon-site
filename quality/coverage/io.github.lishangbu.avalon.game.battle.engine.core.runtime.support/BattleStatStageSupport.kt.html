﻿package io.github.lishangbu.avalon.game.battle.engine.core.runtime.support

import io.github.lishangbu.avalon.game.battle.engine.core.constant.BattleBoostDirectionValues

/**
 * battle stage / boost 规则支持组件。
 *
 * 设计意图：
 * - 把 stage 上下限、倍率公式、别名读取和 boost 合并规则集中到一处；
 * - 避免伤害公式、逃跑公式、条件判断、mutation 落盘各自维护一份 `coerceIn(-6, 6)`；
 * - 明确“能力变化”本身属于一套通用规则，而不是每个调用点自己约定。
 *
 * 当前规则：
 * - stage 统一限制在 `-6..+6`；
 * - stage 为 0 时不再在 `boosts` map 中持久化，避免出现无意义脏键；
 * - boost 写回时会清理同族别名并统一改写为短写键。
 */
object BattleStatStageSupport {
    const val MIN_STAGE: Int = -6
    const val MAX_STAGE: Int = 6
    private const val STANDARD_STAGE_BASE: Double = 2.0
    private const val ACCURACY_EVASION_STAGE_BASE: Double = 3.0

    /**
     * 读取某项能力当前 stage。
     *
     * 返回值永远已经过标准上下限收口，便于调用方直接拿去参与伤害或条件计算。
     */
    fun readStage(
        boosts: Map<String, Int>,
        statId: String,
    ): Int = BattleStatAliasResolver.readValue(boosts, statId)?.let(::normalizeStage) ?: 0

    /**
     * 把原始 stage 值收口到合法范围。
     */
    fun normalizeStage(stage: Int): Int = stage.coerceIn(MIN_STAGE, MAX_STAGE)

    /**
     * 应用一次 stage 增减。
     *
     * 这是 `boost` mutation 真正落盘时应使用的规则：
     * - 先读取当前 stage；
     * - 再叠加增减量；
     * - 最后统一做上下限裁剪。
     */
    fun applyDelta(
        currentStage: Int,
        delta: Int,
    ): Int = normalizeStage(currentStage + delta)

    /**
     * 计算某个 stage 对应的乘区倍率。
     *
     * 规则与主线对战能力阶级一致：
     * - 正向 stage: `(2 + n) / 2`
     * - 负向 stage: `2 / (2 - n)`
     */
    fun stageMultiplier(stage: Int): Double {
        val normalized = normalizeStage(stage)
        return if (normalized >= 0) {
            (STANDARD_STAGE_BASE + normalized) / STANDARD_STAGE_BASE
        } else {
            STANDARD_STAGE_BASE / (STANDARD_STAGE_BASE - normalized)
        }
    }

    /**
     * 计算命中率 / 闪避率阶段对应的倍率。
     *
     * 规则与主线对战一致：
     * - 正向 stage: `(3 + n) / 3`
     * - 负向 stage: `3 / (3 - n)`
     *
     * 之所以单独提供，而不是复用普通能力倍率，是因为命中/闪避与攻防速的乘区底数不同。
     */
    fun accuracyEvasionMultiplier(stage: Int): Double {
        val normalized = normalizeStage(stage)
        return if (normalized >= 0) {
            (ACCURACY_EVASION_STAGE_BASE + normalized) / ACCURACY_EVASION_STAGE_BASE
        } else {
            ACCURACY_EVASION_STAGE_BASE / (ACCURACY_EVASION_STAGE_BASE - normalized)
        }
    }

    /**
     * 规范化一份 boost delta 负载。
     *
     * 使用场景：
     * - `on_boost` hook 改写 relay 后，需要把不同写法的同义键先合并；
     * - 这里只做“键归一 + 数值相加”，不做最终 stage 上下限裁剪；
     * - 真正裁剪发生在写回单位状态时。
     */
    fun normalizeBoostPayload(boosts: Map<String, Int>): Map<String, Int> {
        val normalized = linkedMapOf<String, Int>()
        boosts.forEach { (statId, delta) ->
            val canonicalStatId = BattleStatAliasResolver.canonicalBoostId(statId)
            val mergedDelta = (normalized[canonicalStatId] ?: 0) + delta
            if (mergedDelta == 0) {
                normalized.remove(canonicalStatId)
            } else {
                normalized[canonicalStatId] = mergedDelta
            }
        }
        return normalized.toMap()
    }

    /**
     * 规范化当前单位已经持有的 boosts。
     *
     * 说明：
     * - 会把长短写别名合并成统一短写；
     * - 会把超界旧数据重新裁剪回合法范围；
     * - 结果为 0 的项会被移除。
     */
    fun normalizeStoredBoosts(boosts: Map<String, Int>): Map<String, Int> {
        val normalized = linkedMapOf<String, Int>()
        boosts.forEach { (statId, stage) ->
            val canonicalStatId = BattleStatAliasResolver.canonicalBoostId(statId)
            val normalizedStage = normalizeStage(stage)
            if (normalizedStage == 0) {
                return@forEach
            }
            normalized[canonicalStatId] = normalizedStage
        }
        return normalized.toMap()
    }

    /**
     * 计算本次 boost 请求在当前阶段下“实际还能生效多少”。
     *
     * 例子：
     * - 当前 `atk = 6`，请求 `atk + 1`，结果为空；
     * - 当前 `def = 5`，请求 `def + 3`，结果为 `def + 1`；
     * - 当前 `spe = -5`，请求 `spe - 3`，结果为 `spe - 1`。
     */
    fun resolveAppliedBoostDeltas(
        currentBoosts: Map<String, Int>,
        requestedBoosts: Map<String, Int>,
    ): Map<String, Int> {
        val normalizedCurrentBoosts = normalizeStoredBoosts(currentBoosts)
        val normalizedRequestedBoosts = normalizeBoostPayload(requestedBoosts)
        val applied = linkedMapOf<String, Int>()
        normalizedRequestedBoosts.forEach { (statId, delta) ->
            val currentStage = readStage(normalizedCurrentBoosts, statId)
            val nextStage = applyDelta(currentStage, delta)
            val actualDelta = nextStage - currentStage
            if (actualDelta != 0) {
                applied[BattleStatAliasResolver.canonicalBoostId(statId)] = actualDelta
            }
        }
        return applied.toMap()
    }

    /**
     * 判断本次 boost 变化的整体方向。
     */
    fun resolveBoostDirection(boosts: Map<String, Int>): String {
        val normalizedBoosts = normalizeBoostPayload(boosts)
        val hasPositive = normalizedBoosts.values.any { delta -> delta > 0 }
        val hasNegative = normalizedBoosts.values.any { delta -> delta < 0 }
        return when {
            hasPositive && hasNegative -> BattleBoostDirectionValues.MIXED
            hasPositive -> BattleBoostDirectionValues.RAISE
            hasNegative -> BattleBoostDirectionValues.LOWER
            else -> BattleBoostDirectionValues.NONE
        }
    }

    /**
     * 判断是否至少存在一项正向能力变化。
     */
    fun hasPositiveBoostChange(boosts: Map<String, Int>): Boolean = normalizeBoostPayload(boosts).values.any { delta -> delta > 0 }

    /**
     * 判断是否至少存在一项负向能力变化。
     */
    fun hasNegativeBoostChange(boosts: Map<String, Int>): Boolean = normalizeBoostPayload(boosts).values.any { delta -> delta < 0 }

    /**
     * 把一组 boost delta 落到当前单位 boosts 上。
     *
     * 说明：
     * - 会先读取当前 stage；
     * - 删除该能力项长短写的旧键，避免 `attack` 和 `atk` 并存；
     * - 结果为 0 时直接移除该键，保持 map 干净。
     */
    fun applyBoostDeltas(
        currentBoosts: Map<String, Int>,
        deltas: Map<String, Int>,
    ): Map<String, Int> {
        val nextBoosts = normalizeStoredBoosts(currentBoosts).toMutableMap()
        resolveAppliedBoostDeltas(nextBoosts, deltas).forEach { (statId, delta) ->
            val currentStage = readStage(nextBoosts, statId)
            BattleStatAliasResolver.equivalentIds(statId).forEach(nextBoosts::remove)
            val nextStage = applyDelta(currentStage, delta)
            if (nextStage != 0) {
                nextBoosts[BattleStatAliasResolver.canonicalBoostId(statId)] = nextStage
            }
        }
        return nextBoosts.toMap()
    }
}

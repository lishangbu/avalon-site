package io.github.lishangbu.avalon.game.service.effect

import io.github.lishangbu.avalon.dataset.api.game.model.DatasetAbility
import io.github.lishangbu.avalon.dataset.api.game.model.DatasetItem
import io.github.lishangbu.avalon.dataset.api.game.model.DatasetMove
import io.github.lishangbu.avalon.dataset.api.game.service.GameAbilityQueryPort
import io.github.lishangbu.avalon.dataset.api.game.service.GameItemQueryPort
import io.github.lishangbu.avalon.dataset.api.game.service.GameMoveQueryPort
import io.github.lishangbu.avalon.game.battle.engine.dsl.EffectDefinition
import org.springframework.stereotype.Service

/**
 * 基于真实数据源的战斗 effect 智能导入服务。
 */
@Service
class SmartBattleEffectImportService(
    private val gameMoveQueryPort: GameMoveQueryPort,
    private val gameAbilityQueryPort: GameAbilityQueryPort,
    private val gameItemQueryPort: GameItemQueryPort,
) {
    fun importEffect(effectId: String): EffectDefinition? =
        loadMove(effectId)?.let(::toMoveRecord)?.let(SmartBattleEffectAssembler::fromMove)
            ?: loadAbility(effectId)?.let(::toAbilityRecord)?.let(SmartBattleEffectAssembler::fromAbility)
            ?: loadItem(effectId)?.let(::toItemRecord)?.let(SmartBattleEffectAssembler::fromItem)

    private fun loadMove(internalName: String): DatasetMove? = gameMoveQueryPort.findMoveByInternalName(internalName)

    private fun loadAbility(internalName: String): DatasetAbility? = gameAbilityQueryPort.findAbilityByInternalName(internalName)

    private fun loadItem(internalName: String): DatasetItem? = gameItemQueryPort.findItemByInternalName(internalName)

    private fun toMoveRecord(move: DatasetMove): MoveImportRecord =
        MoveImportRecord(
            internalName = move.internalName,
            name = move.name,
            typeInternalName = move.typeInternalName,
            damageClassInternalName = move.damageClassInternalName,
            targetInternalName = move.targetInternalName,
            accuracy = move.accuracy,
            effectChance = move.effectChance,
            pp = move.pp,
            priority = move.priority,
            power = move.power,
            shortEffect = move.shortEffect,
            effect = move.effect,
            ailmentInternalName = move.ailmentInternalName,
            ailmentChance = move.ailmentChance,
            healing = move.healing,
            drain = move.drain,
        )

    private fun toAbilityRecord(ability: DatasetAbility): AbilityImportRecord =
        AbilityImportRecord(
            internalName = ability.internalName,
            name = ability.name,
            effect = ability.effect,
            introduction = ability.introduction,
        )

    private fun toItemRecord(item: DatasetItem): ItemImportRecord =
        ItemImportRecord(
            internalName = item.internalName,
            name = item.name,
            shortEffect = item.shortEffect,
            effect = item.effect,
            text = item.text,
            attributeInternalNames = item.attributeInternalNames,
            flingEffectInternalName = item.flingEffectInternalName,
        )
}

 * 设计意图：
 * - 提供文档中约定的默认 actor 常量。
 * - 避免在 DSL 节点构造时重复手写字符串。
 */

    val TARGET: ActorId = ActorId("target")

    val FIELD: ActorId = ActorId("field")
    val SIDE: ActorId = ActorId("side")
    val FOE_SIDE: ActorId = ActorId("foe_side")
}

package io.github.lishangbu.avalon.dataset.model

/** 能力值计算器中的单项能力计算结果。 */
data class StatCalculatorEntryResultView(
    /** 能力项 ID。 */
    val statId: String,
    /** 标准化后的能力项内部名称。 */
    val statInternalName: String,
    /** 能力项显示名称。 */
    val statName: String,
    /** 本次计算使用的基础能力值。 */
    val baseStat: Int,
    /** 本次计算使用的个体值。 */
    val iv: Int,
    /** 本次计算使用的努力值。 */
    val ev: Int,
    /** 按当前输入计算出的实际能力值。 */
    val actualValue: Int,
    /** 在当前等级、努力值和性格下的最小能力值。 */
    val minimumValue: Int,
    /** 在当前等级、努力值和性格下的最大能力值。 */
    val maximumValue: Int,
    /** 性格修正系数，使用 90、100、110 表示。 */
    val natureModifier: Int,
)
